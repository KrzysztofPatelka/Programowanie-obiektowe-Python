CREATE TABLE student (id integer PRIMARY KEY AUTOINCREMENT, imie text NOT NULL, nazwisko text NOT NULL);
CREATE TABLE ocena (id integer PRIMARY KEY AUTOINCREMENT, ids integer NOT NULL, ocena integer NOT NULL, FOREIGN KEY (ids) REFERENCES student(id));
CREATE TABLE dziennik (id integer PRIMARY KEY AUTOINCREMENT, nazwa text NOT NULL);
CREATE TABLE dzienniki (id integer PRIMARY KEY AUTOINCREMENT, idd integer NOT NULL, ids NOT NULL, FOREIGN KEY (idd) REFERENCES dziennik(id), FOREIGN KEY (ids) REFERENCES student(id));
